\section{導入}

\begin{frame}{研究の動機}
  \begin{itemize}
    \item 日本語を含む beamer スライドを LuaLaTeX で作成する。
    \item テーマは \alert{metropolis} を用いる。
    \item 数式も通常通り使える：
    \[
      \sqrt{n}(\hat{\theta}-\theta_0)\darrow \Normal(0,\Var(X)).
    \]
  \end{itemize}
\end{frame}
